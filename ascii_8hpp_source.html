60  template<typename T, std::size_t size>
61  void write(std::span<T, size> data) {
62  using std::min;
63  std::size_t count = 0;
64  while (count < data.size()) {
65  const auto num_entries = min(_opts.entries_per_line, data.size() - count);
66  this->_write_formatted(count > 0 ? "\n" : "");
67  this->_write_formatted(_opts.line_prefix);
68  _write(data.data() + count, num_entries);
69  count += num_entries;
70  }
71  }
72 
73  private:
74  template<typename T>
75  void _write(const T* data, std::size_t num_entries) {
76  using PrintType = typename Encoding::Detail::AsciiPrintType<T>::type;
77  std::for_each_n(data, num_entries, [&] (const T& value) {
78  this->_write_formatted(static_cast<PrintType>(value));
79  this->_write_formatted(_opts.delimiter);
