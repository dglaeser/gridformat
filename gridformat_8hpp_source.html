75template<typename Format, typename Opts>
76struct FormatWithOptions {
77 using Options = Opts;
78 Options opts = {};
79
81 constexpr Format operator()(Options _opts) const {
82 return with(std::move(_opts));
83 }
84
86 constexpr Format with(Options _opts) const {
87 Format f;
88 f.opts = std::move(_opts);
89 return f;
90 }
91};

94template<typename VTKFormat>
95struct VTKXMLFormatBase : public FormatWithOptions<VTKFormat, VTK::XMLOptions> {
97 constexpr auto with_encoding(VTK::XML::Encoder e) const {
98 auto cur_opts = this->opts;
99 Variant::unwrap_to(cur_opts.encoder, e);
100 return this->with(std::move(cur_opts));
101 }
102
104 constexpr auto with_data_format(VTK::XML::DataFormat f) const {
105 auto cur_opts = this->opts;
106 Variant::unwrap_to(cur_opts.data_format, f);
107 return this->with(std::move(cur_opts));
108 }
109
111 constexpr auto with_compression(VTK::XML::Compressor c) const {
112 auto cur_opts = this->opts;
113 Variant::unwrap_to(cur_opts.compressor, c);
114 return this->with(std::move(cur_opts));
115 }
116};

305struct TimeSeriesClosure {
306 template<typename Format>
307 constexpr auto operator()(const Format& f) const {
308 if constexpr (Detail::IsVTKXMLFormat<Format>::value)
309 return VTKXMLTimeSeries<Format>{f.opts};
310#if GRIDFORMAT_HAVE_HIGH_FIVE
311 else if constexpr (std::same_as<VTKHDFImage, Format>)
312 return VTKHDFImageTransient{};
313 else if constexpr (std::same_as<VTKHDFUnstructured, Format>)
314 return VTKHDFUnstructuredTransient{};
315 else if constexpr (std::same_as<VTKHDF, Format>)
316 return VTKHDFTransient{};
317#endif // GRIDFORMAT_HAVE_HIGH_FIVE
318 else {
319 static_assert(
320 Detail::always_false<Format>,
321 "Cannot create a time series variant for the given format. This means the format does not "
322 "define a variant for time series, or, the selected format is already time series format."
323 );
324 }
325 }
326};

568template<Concepts::Grid G>
569constexpr auto default_for() {
570 if constexpr (Concepts::ImageGrid<G>)
571 return vti;
572 else if constexpr (Concepts::RectilinearGrid<G>)
573 return vtr;
574 else if constexpr (Concepts::StructuredGrid<G>)
575 return vts;
576 else if constexpr (Concepts::UnstructuredGrid<G>)
577 return vtu;
578 else
579 static_assert(APIDetail::False<G>::value, "Cannot deduce a default format for the given grid");
580}
581
587template<Concepts::Grid G>
588constexpr auto default_for(const G&) {
589 return default_for<G>();
590}
