87  std::ranges::view auto ranks_below(const Location& loc, unsigned direction) const {
88  return std::views::filter(

98  );

107 template<typename T, std::size_t dim>
108 class StructuredGridMapperHelper {
109  public:
110  using Origin = std::array<T, dim>;
111  using Basis = std::array<Origin, dim>;
112 
113  explicit StructuredGridMapperHelper(std::integral auto ranks,
114  const Basis& basis,
115  T default_epsilon = 1e-6)
116  : _basis(basis)
117  , _origins(ranks)
118  , _set(ranks, false)
119  , _default_epsilon{default_epsilon} {
120  std::ranges::fill(_reverse, false);
121  }
122 
123  void reverse(int direction) {
124  _reverse[direction] = !_reverse[direction];
125  }
126 
127  void set_origin_for(int rank, Origin origin) {
128  if (_set[rank])
129  throw ValueError("Origin for given rank already set");
130  _origins[rank] = std::move(origin);
131  _set[rank] = true;
132  }
133 
134  auto make_mapper() const {
135  if (_origins.empty())
136  throw InvalidState("No origins have been set");
137 
138  std::vector<std::array<std::size_t, dim>> map(_origins.size());
139  for (unsigned dir = 0; dir < dim; ++dir) {
140  auto ordinates = _get_ordinates(dir);
141  const auto eps = _epsilon(ordinates);
142  _sort_ordinates(ordinates, eps, _reverse[dir]);
143 
144  if (ordinates.size() > 1) {
145  for (unsigned rank = 0; rank < _origins.size(); ++rank) {
146  const auto rank_origin_ordinate = _get_ordinate(_origins[rank], dir);
147  auto it = std::ranges::find_if(ordinates, [&] (const T o) {
148  using std::abs;
149  return abs(o - rank_origin_ordinate) < eps;
150  });
151  if (it == ordinates.end())
152  throw InvalidState("Could not determine rank ordinate");
153  map[rank][dir] = std::distance(ordinates.begin(), it);
154  }
155  } else {
156  for (unsigned rank = 0; rank < _origins.size(); ++rank)
157  map[rank][dir] = 0;
158  }
159  }
160  return StructuredGridMapper<dim>{std::move(map)};
161  }
162 
163  Origin compute_origin() const {
164  if (_origins.empty())
165  throw InvalidState("No origins have been set");
166 
167  Origin result;
168  for (unsigned dir = 0; dir < dim; ++dir) {
169  auto ordinates = _get_ordinates(dir);
170  _sort_ordinates(ordinates, _reverse[dir]);
171  result[dir] = ordinates[0];
172  }
173  return result;
174  }
175 
176  private:
177  auto _get_ordinates(unsigned int axis) const {
178  std::vector<T> result;
179  result.reserve(_origins.size());
180  std::ranges::copy(
181  std::views::transform(_origins, [&] (const Origin& o) {
182  return _get_ordinate(o, axis);
183  }),
184  std::back_inserter(result)
185  );
186  return result;
187  }
188 
189  auto _get_ordinate(const Origin& o, unsigned int axis) const {
190  return dot_product(o, _basis[axis]);
191  }
192 
193  void _sort_ordinates(std::vector<T>& ordinates, bool reverse) const {
194  if (ordinates.size() > 1)
195  _sort_ordinates(ordinates, _epsilon(ordinates), reverse);
196  }
197 
198  void _sort_ordinates(std::vector<T>& ordinates, const T& eps, bool reverse) const {
199  if (ordinates.size() > 1) {
200  auto [it, _] = Ranges::sort_and_unique(ordinates, {}, [&] (T a, T b) {
201  using std::abs;
202  return abs(a - b) < eps;
203  });
204  ordinates.erase(it, ordinates.end());
205  if (reverse)
206  std::ranges::reverse(ordinates);
207  }
208  }
209 
210  T _epsilon(const std::vector<T>& ordinates) const {
211  const auto max = *std::ranges::max_element(ordinates);
212  const auto min = *std::ranges::min_element(ordinates);
213  const auto size = max - min;
214 
215  if (size > _default_epsilon) {
216  std::vector<T> dx; dx.reserve(ordinates.size());
217  std::adjacent_difference(ordinates.begin(), ordinates.end(), std::back_inserter(dx));
218  std::ranges::for_each(dx, [] (auto& v) { v = std::abs(v); });
219  std::ranges::sort(dx);
220  for (T _dx : dx)
221  if (_dx > 1e-8*size)
222  return _dx*0.1;
223  }
224 
225  return _default_epsilon;
226  }
227 
228  const Basis& _basis;
229  std::vector<Origin> _origins;
230  std::vector<bool> _set;
231  std::array<bool, dim> _reverse;
232  T _default_epsilon;
233 };

237 class StructuredParallelGridHelper {
238  public:
239  StructuredParallelGridHelper(const Communicator& comm, int root_rank = 0)
240  : _comm{comm}
241  , _root_rank{root_rank}
242  {}

244  template<Concepts::Scalar CT,
245  std::size_t dim,
246  Concepts::StaticallySizedMDRange<2> B = std::array<std::array<CT, dim>, dim>>
247  auto compute_extents_and_origin(const std::vector<CT>& all_origins,
248  const std::vector<std::size_t>& all_extents,
249  const std::array<bool, dim>& is_negative_axis,
250  const B& basis = GridDetail::standard_basis<CT, dim>()) const {
251  const auto num_ranks = Parallel::size(_comm);
252  std::vector<std::array<std::size_t, dim>> pieces_begin(num_ranks);
253  std::vector<std::array<std::size_t, dim>> pieces_end(num_ranks);
254  std::array<std::size_t, dim> whole_extent;
255  std::array<CT, dim> origin;
256 
257  if (Parallel::rank(_comm) == _root_rank) {
258  const auto default_epsilon = 1e-6*std::ranges::max(all_origins | std::views::transform([] (CT v) {
259  using std::abs;
260  return abs(v);
261  }));
262  const auto mapper_helper = _make_mapper_helper(basis, all_origins, is_negative_axis, default_epsilon);
263  const auto rank_mapper = mapper_helper.make_mapper();
264  origin = mapper_helper.compute_origin();
265 
266  for (unsigned dir = 0; dir < dim; ++dir) {
267  for (int rank = 0; rank < num_ranks; ++rank) {
268  auto ranks_below = rank_mapper.ranks_below(rank_mapper.location(rank), dir);
269  const std::size_t offset = std::accumulate(
270  std::ranges::begin(ranks_below),
271  std::ranges::end(ranks_below),
272  std::size_t{0},
273  [&] (const std::size_t current, int r) {
274  return current + Parallel::access_gathered<dim>(all_extents, _comm, {dir, r});
275  }
276  );
277  pieces_begin[rank][dir] = offset;
278  pieces_end[rank][dir] = offset + Parallel::access_gathered<dim>(all_extents, _comm, {dir, rank});
279  }
280 
281  whole_extent[dir] = (*std::max_element(
282  pieces_end.begin(), pieces_end.end(),
283  [&] (const auto& a1, const auto& a2) { return a1[dir] < a2[dir]; }
284  ))[dir];
285  }
286  }
287 
288  return std::make_tuple(
289  std::move(pieces_begin),
290  std::move(pieces_end),
291  std::move(whole_extent),
292  std::move(origin)
293  );
294  }
295 
296  private:
297  template<Concepts::StaticallySizedMDRange<2> Basis, Concepts::Scalar CT, std::size_t dim>
298  auto _make_mapper_helper(const Basis& basis,
299  const std::vector<CT>& all_origins,
300  const std::array<bool, dim>& is_negative_axis,
301  CT default_eps) const {
302  StructuredGridMapperHelper<CT, dim> helper(Parallel::size(_comm), basis, default_eps);
303  std::ranges::for_each(Parallel::ranks(_comm), [&] (int rank) {
304  helper.set_origin_for(rank, Parallel::access_gathered<dim>(all_origins, _comm, rank));
305  });
306 
307  for (unsigned dir = 0; dir < dim; ++dir)
308  if (is_negative_axis[dir])
309  helper.reverse(dir);
310 
311  return helper;
312  }
313 
314  const Communicator& _comm;
315  int _root_rank;
316 };
