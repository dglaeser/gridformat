68  virtual void _write(const std::string& filename_with_ext) const override {
69  const auto& local_extents = extents(this->grid());
70  const auto [origin, is_negative_axis] = _get_origin_and_orientations();
71 
72  PVTK::StructuredParallelGridHelper helper{_comm};
73  const auto all_origins = Parallel::gather(_comm, origin, root_rank);
74  const auto all_extents = Parallel::gather(_comm, local_extents, root_rank);
75  const auto [exts_begin, exts_end, whole_extent, _] = helper.compute_extents_and_origin(
76  all_origins,
77  all_extents,
78  is_negative_axis
79  );
80 
81  const auto my_whole_extent = Parallel::broadcast(_comm, whole_extent, root_rank);
82  const auto my_extent_offset = Parallel::scatter(_comm, Ranges::flat(exts_begin), root_rank);
83 
84  _write_piece(filename_with_ext, Ranges::to_array<dim>(my_extent_offset), {my_whole_extent});
85  Parallel::barrier(_comm); // ensure all pieces finished successfully
86  if (Parallel::rank(_comm) == 0)
87  _write_pvtr_file(filename_with_ext, my_whole_extent, exts_begin, exts_end);
88  Parallel::barrier(_comm); // ensure .pvtr file is written before returning
89  }

91  auto _get_origin_and_orientations() const {
92  std::array<CT, dim> origin;
93  std::array<bool, dim> is_negative_axis;
94  for (unsigned dir = 0; dir < dim; ++dir) {
95  std::array<CT, 2> ordinates_01{0, 0};
96  const auto& dir_ordinates = ordinates(this->grid(), dir);
97  std::ranges::copy(std::views::take(dir_ordinates, 2), ordinates_01.begin());
98  origin[dir] = ordinates_01[0];
99  is_negative_axis[dir] = ordinates_01[1] - ordinates_01[0] < CT{0};
100  }
101  return std::make_tuple(origin, is_negative_axis);
102  }

114  void _write_pvtr_file(const std::string& filename_with_ext,
115  const std::array<std::size_t, dim>& extents,
116  const std::vector<std::array<std::size_t, dim>>& proc_extents_begin,

118  std::ofstream file_stream(filename_with_ext, std::ios::out);
119 
120  XMLElement pvtk_xml("VTKFile");
121  pvtk_xml.set_attribute("type", "PRectilinearGrid");
122 
123  XMLElement& grid = pvtk_xml.add_child("PRectilinearGrid");
124  grid.set_attribute("WholeExtent", VTK::CommonDetail::extents_string(extents));
125 
126  XMLElement& ppoint_data = grid.add_child("PPointData");
127  XMLElement& pcell_data = grid.add_child("PCellData");
128  XMLElement& pcoords = grid.add_child("PCoordinates");
129  std::visit([&] (const auto& encoder) {
130  std::visit([&] (const auto& data_format) {
131  PVTK::PDataArrayHelper pdata_helper{encoder, data_format, ppoint_data};
132  PVTK::PDataArrayHelper cdata_helper{encoder, data_format, pcell_data};
133  std::ranges::for_each(this->_point_field_names(), [&] (const std::string& name) {
134  pdata_helper.add(name, this->_get_point_field(name));
135  });
136  std::ranges::for_each(this->_cell_field_names(), [&] (const std::string& name) {
137  cdata_helper.add(name, this->_get_cell_field(name));
138  });
139 
140  std::visit([&] <typename T> (const Precision<T>& prec) {
141  for (unsigned int i = 0; i < space_dim; ++i) {
142  XMLElement& pdata_array = pcoords.add_child("PDataArray");
143  pdata_array.set_attribute("NumberOfComponents", "1");
144  pdata_array.set_attribute("Name", "X_" + std::to_string(i));
145  pdata_array.set_attribute("format", VTK::data_format_name(encoder, data_format));
146  pdata_array.set_attribute("type", VTK::attribute_name(prec));
147  }
148  }, this->_xml_settings.coordinate_precision);
149  }, this->_xml_settings.data_format);
150  }, this->_xml_settings.encoder);
151 
152  std::ranges::for_each(Parallel::ranks(_comm), [&] (int rank) {
153  std::array<std::size_t, dim> extents_begin;
154  std::array<std::size_t, dim> extents_end;
155  for (unsigned dir = 0; dir < dim; ++dir) {
156  extents_begin[dir] = proc_extents_begin[rank][dir];
157  extents_end[dir] = proc_extents_end[rank][dir];
158  }
159  auto& piece = grid.add_child("Piece");
160  piece.set_attribute("Extent", VTK::CommonDetail::extents_string(extents_begin, extents_end));
161  piece.set_attribute("Source", PVTK::piece_basefilename(filename_with_ext, rank) + ".vtr");
162  });
163 
164  write_xml_with_version_header(pvtk_xml, file_stream, Indentation{{.width = 2}});
165  }
166 };
