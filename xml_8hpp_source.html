124  template<typename GridCoordinateType>
125  static XMLSettings from(const XMLOptions& opts) {
126  const auto _enc = _make_encoder(opts.encoder);
127  const auto _format = _make_data_format(_enc, opts.data_format);
128  const auto _comp = _make_compressor(_enc, opts.compressor);
129  return {
130  .encoder = _enc,
131  .compressor = _comp,
132  .data_format = _format,
133  .coordinate_precision =
134  Variant::is<Automatic>(opts.coordinate_precision) ?
135  XML::CoordinatePrecision{Precision<GridCoordinateType>{}} :
136  Variant::without<Automatic>(opts.coordinate_precision),
137  .header_precision = opts.header_precision
138  };
139  }
140 
141  private:
142  static XML::Encoder _make_encoder(const typename XMLOptions::EncoderOption& enc) {
143  if (Variant::is<Automatic>(enc))
144  return Encoding::base64;
145  return Variant::without<Automatic>(enc);
146  }
147 
148  static XML::DataFormat _make_data_format(const typename XML::Encoder& enc,
149  const typename XMLOptions::DataFormatOption& data_format) {
150  if (Variant::is<Automatic>(data_format))
151  return Variant::is<Encoding::Ascii>(enc)
152  ? XML::DataFormat{VTK::DataFormat::inlined}
153  : XML::DataFormat{VTK::DataFormat::appended};
154  return Variant::without<Automatic>(data_format);
155  }
156 
157  static XML::Compressor _make_compressor(const typename XML::Encoder& enc,
158  const typename XMLOptions::CompressorOption& compressor) {
159  if (Variant::is<Automatic>(compressor))
160  return Variant::is<Encoding::Ascii>(enc)
161  ? XML::Compressor{none}
162  : XML::Compressor{XML::DefaultCompressor{}};
163  if (Variant::is<Encoding::Ascii>(enc) && !Variant::is<None>(compressor)) {
164  log_warning("Ascii output cannot be compressed. Ignoring chosen compressor...");
165  return none;
166  }
167  return Variant::without<Automatic>(compressor);
168  }
169  };

274  .get_child("FieldData");
