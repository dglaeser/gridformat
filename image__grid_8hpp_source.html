40  class Iterator : public ForwardIteratorFacade<Iterator, IndexTuple, const IndexTuple&> {
41  public:
42  Iterator() = default;
43  Iterator(const IndexTuple& bounds)
44  : _bounds{&bounds}
45  , _current{} {
46  std::ranges::fill(_current, std::size_t{0});
47  }
48 
49  Iterator(const IndexTuple& bounds, bool)
50  : Iterator(bounds) {
51  _current[0] = (*_bounds)[0];
52  }
53 
54  private:
55  friend IteratorAccess;
56 
57  const IndexTuple& _dereference() const {
58  return _current;
59  }
60 
61  bool _is_equal(const Iterator& other) const {
62  if (_bounds != other._bounds)
63  return false;
64  return std::ranges::equal(_current, other._current);
65  }
66 
67  void _increment() {
68  assert(_is_valid());
69  for (int dir = dim - 1; dir > 0; dir--) {
70  if (_current[dir]++; _current[dir] == (*_bounds)[dir])
71  _current[dir] = 0;
72  else
73  return;
74  }
75  _current[0]++;
76  }
77 
78  bool _is_valid() const {
79  for (unsigned i = 0; i < dim; ++i)
80  if (_current[i] >= (*_bounds)[i])
81  return false;
82  return true;
83  }
84 
85  const IndexTuple* _bounds{nullptr};
86  IndexTuple _current;
87  };
88

91  explicit MDIndexRange(Bounds&& bounds)
92  : _bounds(Ranges::to_array<dim, std::size_t>(std::forward<Bounds>(bounds)))
93  {}
94 
95  auto begin() const { return Iterator{_bounds}; }
96  auto end() const { return Iterator{_bounds, true}; }

99  std::size_t size() const {

176  ImageGrid(std::array<CoordinateType, dim> origin,
177  std::array<CoordinateType, dim> size,
178  std::array<std::size_t, dim> cells)
179  : _lower_right{std::move(origin)}
180  , _upper_right{Ranges::apply_pairwise<CoordinateType>(std::plus{}, _lower_right, size)}
181  , _spacing{_compute_spacing(cells)}
182  , _cell_index_tuples{cells}
183  , _point_index_tuples{Ranges::incremented(cells, 1)}
184  , _cell_point_offsets{Ranges::filled_array<dim, std::size_t>(2)}
185  , _point_mapper{Ranges::incremented(cells, 1)} {
186  if (std::ranges::any_of(cells, [] (const std::integral auto i) { return i == 0; }))
187  throw ValueError("Number of cells in each direction must be > 0");
188  }

190  std::size_t number_of_cells() const { return _cell_index_tuples.size(); }
191  std::size_t number_of_points() const { return _point_index_tuples.size(); }
192  std::size_t number_of_cells(unsigned int direction) const { return _cell_index_tuples.size(direction); }
193  std::size_t number_of_points(unsigned int direction) const { return _point_index_tuples.size(direction); }
194  const auto& origin() const { return _lower_right; }
195  const auto& spacing() const { return _spacing; }
196 
198  auto extents() const {
199  return Ranges::to_array<dim, std::size_t>(
200  std::views::iota(std::size_t{0}, dim) | std::views::transform([&] (std::size_t i) {
201  return _cell_index_tuples.size(i);
202  })
203  );
204  }
205 
207  auto ordinates(int direction) const {
208  std::vector<CoordinateType> result(number_of_points(direction));
209  for (unsigned int i = 0; i < result.size(); ++i)
210  result[i] = _ordinate_at(i, direction);
211  return result;
212  }

215  auto position(const Point& p) const {
216  return Ranges::to_array<dim, CoordinateType>(
217  std::views::iota(std::size_t{0}, dim)
218  | std::views::transform([&] (std::size_t direction) {
219  return _ordinate_at(p.location[direction], direction);
220  })
221  );
222  }

225  auto center(const Cell& c) const {
226  unsigned corner_count = 0;
227  auto result = Ranges::filled_array<dim>(CoordinateType{0});
228  for (const auto& p : points(c)) {
229  corner_count++;
230  const auto ppos = position(p);
231  for (unsigned i = 0; i < dim; ++i)
232  result[i] += ppos[i];
233  }
234  std::ranges::for_each(result, [&] (auto& v) { v /= static_cast<CoordinateType>(corner_count); });
235  return result;
236  }

239  std::size_t id(const Point& p) const {
240  return _point_mapper.map(p.location);
241  }

244  std::ranges::range auto cells() const {
245  return _cell_index_tuples | std::views::transform([] (const auto& indices) { return Cell{indices}; });
246  }
247 
249  std::ranges::range auto points() const {
250  return _point_index_tuples | std::views::transform([] (const auto& indices) { return Point{indices}; });
251  }
252 
254  std::ranges::range auto points(const Cell& cell) const {
255  return _cell_point_offsets
256  | std::views::transform([loc = cell.location] (const auto& offset) {
257  auto point_loc = loc;
258  std::transform(
259  point_loc.begin(), point_loc.end(), offset.begin(), point_loc.begin(),
260  [] (std::size_t p, std::size_t o) { return p + o; }
261  );
262  return Point{std::move(point_loc)};
263  });
264  }
265 
266  private:
267  CoordinateType _ordinate_at(std::size_t i, int direction) const {
268  return _lower_right[direction] + _spacing[direction]*static_cast<CoordinateType>(i);
269  }
270 
271  template<Concepts::StaticallySizedMDRange<1> Cells>
272  auto _compute_spacing(Cells&& cells) const {
273  return Ranges::apply_pairwise(
274  std::divides{},
275  Ranges::apply_pairwise(std::minus{}, _upper_right, _lower_right),
276  cells
277  );
278  }
279 
280  std::array<unsigned int, dim> _directions_in_descending_size() const {
281  std::array<unsigned int, dim> result;
282  for (int i = 0; i < dim; ++i)
283  result[i] = i;
284  std::ranges::sort(result, [&] (unsigned int dir1, unsigned int dir2) {
285  return _cell_index_tuples.size(dir1) < _cell_index_tuples.size(dir2);
286  });
287  return result;
288  }

290  std::array<CoordinateType, dim> _lower_right;
291  std::array<CoordinateType, dim> _upper_right;
292  std::array<CoordinateType, dim> _spacing;
293  Detail::MDIndexRange<dim> _cell_index_tuples;
294  Detail::MDIndexRange<dim> _point_index_tuples;
295  Detail::MDIndexRange<dim> _cell_point_offsets;
296  FlatIndexMapper<dim> _point_mapper;
