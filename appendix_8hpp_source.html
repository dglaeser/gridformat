128 void write_xml_element_with_offsets(const XMLElement& e,
129 std::ostream& s,
130 Indentation& ind,
131 std::vector<std::size_t>& offset_positions) {
132 const std::string empty_string_for_max_header(
133 std::to_string(std::numeric_limits<std::size_t>::max()).size(),
134 ' '
135 );
136
137 const auto cache_offset_xml_pos = [&] () {
138 s << " offset=\"";
139 offset_positions.push_back(s.tellp());
140 s << empty_string_for_max_header;
141 s << "\"";
142 };
143
144 if (!e.has_content() && e.number_of_children() == 0) {
145 s << ind;
146 GridFormat::XML::Detail::write_xml_tag_open(e, s, "");
147 if (e.name() == "DataArray")
148 cache_offset_xml_pos();
149 s << "/>";
150 } else {
151 s << ind;
152 GridFormat::XML::Detail::write_xml_tag_open(e, s);
153 if (e.name() == "DataArray")
154 cache_offset_xml_pos();
155 s << "\n";
156
157 if (e.has_content()) {
158 e.stream_content(s);
159 s << "\n";
160 }
161
162 ++ind;
163 for (const auto& c : children(e)) {
164 write_xml_element_with_offsets(c, s, ind, offset_positions);
165 s << "\n";
166 }
167 --ind;
168
169 s << ind;
170 GridFormat::XML::Detail::write_xml_tag_close(e, s);
171 }

182 template<typename Context, typename Encoder>
183 requires(!std::is_const_v<std::remove_reference_t<Context>>)
184 inline void write_with_appendix(Context&& context,
185 std::ostream& s,
186 const Encoder& encoder,
187 Indentation indentation = {}) {
188 if (produces_valid_xml(encoder))
189 s << "<?xml version=\"1.0\"?>\n";
190
191 AppendixStreamObserver observer;
192 context.appendix.set_observer(&observer);
193
194 auto& app_element = context.xml_representation.add_child("AppendedData");
195 app_element.set_attribute("encoding", attribute_name(encoder));
196 app_element.set_content(Detail::XMLAppendixContent{context.appendix});
197
198 const auto offset_positions = write_xml_element_with_offsets(
199 context.xml_representation, s, indentation
200 );
201 const auto& offsets = observer.offsets();
202 if (offsets.size() != offset_positions.size())
203 throw SizeError("Number of written & registered offsets does not match");
204
205 const auto cur_pos = s.tellp();
206 for (std::size_t i = 0; i < offsets.size(); ++i) {
207 s.seekp(offset_positions[i]);
208 const std::string offset_str = std::to_string(offsets[i]);
209 s.write(offset_str.data(), offset_str.size());
210 }
211 s.seekp(cur_pos);
212 }
