196  app_element.set_content(Detail::XMLAppendixContent{context.appendix});

198  const auto offset_positions = write_xml_element_with_offsets(
199  context.xml_representation, s, indentation
200  );
201  const auto& offsets = observer.offsets();
202  if (offsets.size() != offset_positions.size())
203  throw SizeError("Number of written & registered offsets does not match");
204 
205  const auto cur_pos = s.tellp();
206  for (std::size_t i = 0; i < offsets.size(); ++i) {
207  s.seekp(offset_positions[i]);
208  const std::string offset_str = std::to_string(offsets[i]);
209  s.write(offset_str.data(), offset_str.size());
210  }
211  s.seekp(cur_pos);
212  }
