35concept GridFactory = requires(T& t) {
36 typename T::ctype;
37 { t.insert_point(std::declval<const std::array<typename T::ctype, space_dim>&>()) };
38 { t.insert_cell(GridFormat::CellType{}, std::declval<const std::vector<std::size_t>&>()) };

166 template<std::size_t space_dim = 3, Concepts::GridFactory<space_dim> Factory>
167 void export_grid(Factory& factory) const {
168 if (number_of_points() > 0) {
169 const auto point_field = points();
170 const auto point_layout = point_field->layout();
171 const auto read_space_dim = point_layout.extent(1);
172 const auto copied_space_dim = std::min(space_dim, read_space_dim);
173 if (point_layout.extent(0) != number_of_points()) {
174 std::ostringstream s; s << point_layout;
175 throw SizeError(
176 "Point layout " + s.str() + " does not match number of points: " + std::to_string(number_of_points())
177 );
178 }
179
180 point_field->visit_field_values([&] <typename T> (std::span<const T> coords) {
181 auto p = Ranges::filled_array<space_dim>(typename Factory::ctype{0.0});
182 for (std::size_t i = 0; i < point_layout.extent(0); ++i) {
183 for (std::size_t dir = 0; dir < copied_space_dim; ++dir)
184 p[dir] = static_cast<typename Factory::ctype>(coords[i*read_space_dim + dir]);
185 factory.insert_point(std::as_const(p));
186 }
187 });
188 }
189 visit_cells([&] (GridFormat::CellType ct, const std::vector<std::size_t>& corners) {
190 factory.insert_cell(std::move(ct), corners);
191 });
192 }
