155  template<Concepts::Scalar T>
156  void _write_values(HDF5File& file,
157  const DataSetPath& path,
158  const std::vector<T>& values,
159  const IOContext& context) const {
160  if (context.is_parallel) {
161  const auto num_values = values.size();
162  const auto total_num_values = Parallel::sum(_comm, num_values, root_rank);
163  const auto my_total_num_values = Parallel::broadcast(_comm, total_num_values, root_rank);
164  const auto my_offset = _accumulate_rank_offset(num_values);
165  DataSetSlice slice{
166  .size = std::vector{my_total_num_values},
167  .offset = std::vector{my_offset},
168  .count = std::vector{num_values}
169  };
170  file.write(values, path, slice);
171  } else {
172  file.write(values, path);
173  }
174  }

210  template<bool is_point_field, typename FieldValues>
211  void _write_field(HDF5File& file,
212  const DataSetPath& path,
213  const FieldValues& values,
214  const IOContext& context) const {
215  if (context.is_parallel) {
216  const auto layout = get_md_layout(values);
217  std::vector<std::size_t> count(layout.dimension());
218  layout.export_to(count);
219 
220  std::vector<std::size_t> size = count;
221  std::vector<std::size_t> offset(layout.dimension(), 0);
222  offset.at(0) = is_point_field ? context.my_point_offset : context.my_cell_offset;
223  size.at(0) = is_point_field ? context.num_points_total : context.num_cells_total;
224 
225  file.write(values, path, {
226  .size = size,
227  .offset = offset,
228  .count = count
229  });
230  } else {
231  file.write(values, path);
232  }
233  }

235  template<typename Visitor>
236  void _visit_field_values(const Field& field, const Visitor& visitor) const {
237  field.precision().visit([&] <typename T> (const Precision<T>&) {
238  const auto layout = field.layout();
239  if (layout.dimension() == 1) {
240  std::vector<T> data(layout.extent(0));
241  field.export_to(data);
242  visitor(data);
243  } else if (layout.dimension() == 2) {
244  std::vector<Vector<T>> data(layout.extent(0));
245  field.export_to(data);
246  visitor(data);
247  } else if (layout.dimension() == 3) {
248  std::vector<Tensor<T>> data(layout.extent(0));
249  field.export_to(data);
250  visitor(data);
251  } else {
252  throw NotImplemented("Support for fields with dimension > 3 or < 1");
253  }
254  });
255  }

257  std::size_t _accumulate_rank_offset(std::size_t my_size) const {
258  auto all_offsets = Parallel::gather(_comm, my_size, root_rank);
259  if (Parallel::rank(_comm) == root_rank) {
260  std::partial_sum(all_offsets.begin(), std::prev(all_offsets.end()), all_offsets.begin());
261  std::copy(std::next(all_offsets.rbegin()), all_offsets.rend(), all_offsets.rbegin());
262  all_offsets[0] = 0;
263  }
264  const auto my_offset = Parallel::scatter(_comm, all_offsets, root_rank);
265  if (my_offset.size() != 1)
266  throw ValueError("Unexpected scatter result");
267  return my_offset[0];
268  }
