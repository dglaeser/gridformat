55 void _open(const std::string& filename, typename GridReader::FieldNames& field_names) override {
56 _file = HDF5File{filename, _comm, HDF5File::read_only};
57
58 const auto type = VTKHDF::get_file_type(_file.value());
59 if (type != "UnstructuredGrid")
60 throw ValueError("Incompatible VTK-HDF type: '" + type + "', expected 'UnstructuredGrid'.");
61
62 VTKHDF::check_version_compatibility(_file.value(), {2, 0});
63
64 if (_file->exists("VTKHDF/Steps"))
65 _file->visit_attribute("VTKHDF/Steps/NSteps", [&] (auto&& field) {
66 _num_steps.emplace();
67 field.export_to(_num_steps.value());
68 _step_index.emplace(0);
69 });
70
71 _compute_piece_offsets();
72 const auto copy_names_in = [&] (const std::string& group, auto& storage) {
73 if (_file->exists(group))
74 std::ranges::copy(_file->dataset_names_in(group), std::back_inserter(storage));
75 };
76 copy_names_in("VTKHDF/CellData", field_names.cell_fields);
77 copy_names_in("VTKHDF/PointData", field_names.point_fields);
78 copy_names_in("VTKHDF/FieldData", field_names.meta_data_fields);
79 }

87 void _compute_piece_offsets() {
88 _cell_offset = 0;
89 _point_offset = 0;
90 _num_cells = _read_rank_scalar<std::size_t>("/VTKHDF/NumberOfCells", _get_part_offset());
91 _num_points = _read_rank_scalar<std::size_t>("/VTKHDF/NumberOfPoints", _get_part_offset());
92
93 if (Parallel::size(_comm) > 1) {
94 _check_communicator_size();
95 const auto proc_cells = Parallel::broadcast(_comm, Parallel::gather(_comm, _num_cells));
96 const auto proc_points = Parallel::broadcast(_comm, Parallel::gather(_comm, _num_points));
97 const auto compute_offset = [&] (const auto& v) {
98 return std::accumulate(v.begin(), v.begin() + Parallel::rank(_comm), std::size_t{0});
99 };
100 _cell_offset += compute_offset(proc_cells);
101 _point_offset += compute_offset(proc_points);
102 }
103 }

231 void _check_communicator_size() const {
232 const auto nprocs = _number_of_pieces_in_file();
233 if (nprocs != static_cast<std::size_t>(Parallel::size(_comm)))
234 throw SizeError(
235 "Can only read the file in parallel if the size of the communicator matches the size "
236 "of that used when writing the file. Please read in the file sequentially on one process "
237 "and distribute the grid yourself, or restart the parallel run with "
238 + std::to_string(nprocs) + " processes."

242 void _visit_cells(const typename GridReader::CellVisitor& visitor) const override {
243 const auto num_connectivity_ids = _file.value().template read_dataset_to<std::vector<std::size_t>>(
244 "VTKHDF/NumberOfConnectivityIds",
245 HDF5::Slice{
246 .offset = {_get_part_offset()},
247 .count = {static_cast<std::size_t>(Parallel::size(_comm))}
248 }
249 );
250 const auto my_num_connectivity_ids = num_connectivity_ids.at(Parallel::rank(_comm));
251 const auto my_id_offset = _get_connectivity_id_offset() + std::accumulate(
252 num_connectivity_ids.begin(),
253 num_connectivity_ids.begin() + Parallel::rank(_comm),
254 std::size_t{0}
255 );
256 auto offsets = _file.value().template read_dataset_to<std::vector<std::size_t>>(
257 "VTKHDF/Offsets",
258 HDF5::Slice{
259 // offsets have length num_cells+1, so we need to add +1 per rank to the cell offsets
260 .offset = {
261 _get_cells_offset() + _cell_offset
262 + Parallel::size(_comm)*_step_index.value_or(0) +
263 + Parallel::rank(_comm)
264 },
265 .count = {_num_cells + 1}
266 }
267 );
268 auto types = _file.value().template read_dataset_to<std::vector<std::uint_least8_t>>(
269 "VTKHDF/Types",
270 HDF5::Slice{.offset = {_get_cells_offset() + _cell_offset}, .count = {_num_cells}}
271 );
272 auto connectivity = _file.value().template read_dataset_to<std::vector<std::size_t>>(
273 "VTKHDF/Connectivity",
274 HDF5::Slice{.offset = {my_id_offset}, .count = {my_num_connectivity_ids}}
275 );
276
277 std::vector<std::size_t> corners;
278 for (std::size_t i = 0; i < types.size(); ++i) {
279 assert(offsets.at(i+1) > offsets.at(i));
280 const auto num_corners = offsets.at(i+1) - offsets.at(i);
281 corners.resize(num_corners);
282 for (std::size_t c = 0; c < num_corners; ++c)
283 corners[c] = connectivity.at(offsets.at(i) + c);
284 visitor(VTK::cell_type(types.at(i)), corners);
285 }

323 FieldPtr _meta_data_field(std::string_view name) const override {
324 const auto path = "VTKHDF/FieldData/" + std::string{name};
325 const auto dims = _file.value().get_dimensions(path).value();
326 if (dims.size() == 1)
327 return make_field_ptr(VTKHDF::DataSetField{_file.value(), path});
328 if (dims.size() != 2)
329 throw SizeError("Unexpected field data array size");
330 auto offset = dims;
331 auto count = offset;
332 count.at(0) = 1;
333 std::ranges::fill(offset, 0);
334 offset.at(0) = _get_field_data_offset(name);
335 return make_field_ptr(VTKHDF::DataSetField{
336 _file.value(),
337 MDLayout{count | std::views::drop(1)},
338 _file.value().get_precision(path).value(),
339 [p=path, o=offset, c=count] (const HDF5File& f) {
340 return f.visit_dataset(p, [&] <typename F> (F&& field) {
341 return FlattenedField{make_field_ptr(std::move(field))}.serialized();
342 }, HDF5::Slice{.offset = o, .count = c});
343 }
344 });
345 }

347 template<Concepts::Scalar T>
348 T _read_rank_scalar(const std::string& path, const std::size_t base_offset = 0) const {
349 T result;
350 _file.value().visit_dataset(
351 path,
352 [&] (auto&& field) { field.export_to(result); },
353 HDF5::Slice{
354 .offset = {base_offset + static_cast<std::size_t>(Parallel::rank(_comm))},
355 .count = {1}
356 }
357 );
358 return result;
359 }
360
361 HDF5::Slice _get_slice(const std::string& path,
362 std::size_t count,
363 std::size_t offset) const {
364 auto ds_size = _file.value().get_dimensions(path).value();
365 auto ds_offset = ds_size;
366 std::ranges::fill(ds_offset, std::size_t{0});
367 ds_size.at(0) = count;
368 ds_offset.at(0) = offset;
369 return {.offset = ds_offset, .count = ds_size};
370 }
371
372 auto _serialization_callback(std::string path, std::optional<HDF5::Slice> slice = {}) const {
373 return [_p=std::move(path), _s=std::move(slice)] (const HDF5File& file) {
374 if (_s)
375 return file.visit_dataset(_p, [&] <typename F> (F&& f) { return f.serialized(); }, _s.value());
376 return file.visit_dataset(_p, [&] <typename F> (F&& f) { return f.serialized(); });
377 };
378 }
