50 template<typename Enc, typename Format>
51 std::string data_format_name(const Enc& e, const Format& format) {
52  const std::string encoder_name = attribute_name(e);
53  const std::string format_name = Overload{
54  [] (const DataFormat::Appended&) { return "appended"; },
55  [] (const DataFormat::Inlined&) { return "inlined"; }
56  } (format);
57  const std::string other_format_name = Overload{
58  [] (const DataFormat::Appended&) { return "GridFormat::VTK::inlined"; },
59  [] (const DataFormat::Inlined&) { return "GridFormat::VTK::appended"; }
60  } (format);
61  throw ValueError(
62  "VTK's '" + format_name + "' data format cannot be used with " + encoder_name
63  + " encoding. Please choose '" + other_format_name + "' or a different encoder."
64  );
65 }
