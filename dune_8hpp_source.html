324  inline int dune_to_gfmt_sub_entity(const DUNE::GeometryType& gt, int i, int codim) {
325  if (gt.isTriangle()) {
326  if (codim == 1) {
327  assert(i < 3);
328  static constexpr int map[3] = {0, 2, 1};
329  return map[i];
330  }
331  }
332  if (gt.isQuadrilateral()) {
333  if (codim == 2) {
334  assert(i < 4);
335  static constexpr int map[4] = {0, 1, 3, 2};
336  return map[i];
337  }
338  if (codim == 1) {
339  assert(i < 4);
340  static constexpr int map[4] = {3, 1, 0, 2};
341  return map[i];
342  }
343  }
344  if (gt.isTetrahedron()) {
345  if (codim == 2) {
346  assert(i < 6);
347  static constexpr int map[6] = {0, 2, 1, 3, 4, 5};
348  return map[i];
349  }
350  if (codim == 1) {
351  assert(i < 4);
352  static constexpr int map[4] = {3, 0, 2, 1};
353  return map[i];
354  }
355  }
356  if (gt.isHexahedron()) {
357  if (codim == 3) {
358  assert(i < 8);
359  static constexpr int map[8] = {0, 1, 3, 2, 4, 5, 7, 6};
360  return map[i];
361  }
362  if (codim == 2) {
363  assert(i < 12);
364  static constexpr int map[12] = {8, 9, 11, 10, 3, 1, 0, 2, 7, 5, 4, 6};
365  return map[i];
366  }
367  }
368  return i;
369  }

380  // Exposes the lagrange points of a geometry in gridformat ordering
381  template<typename GridView>
382  class LocalPoints {
383  // reserve space for third-order hexahedra
384  static constexpr std::size_t reserved_size = 64;
385 
386  public:
387  explicit LocalPoints(unsigned int order)
388  : _points(order)
389  {}
390 
391  void build(const DUNE::GeometryType& geo_type) {
392  if (geo_type.dim() != GridView::dimension)
393  throw ValueError("Dimension of given geometry does not match the grid");
394  _points.build(geo_type);
395  _setup_sorted_indices(geo_type);
396  }
397 
398  std::size_t size() const {
399  return _points.size();
400  }
401 
402  const auto& at(std::size_t i) const {
403  return _points[_sorted_indices.at(i)];
404  }
405 
406  std::ranges::range auto get() const {
407  return _sorted_indices | std::views::transform([&] (unsigned int i) {
408  return _points[i];
409  });
410  }
411 
412  private:
413  void _setup_sorted_indices(const DUNE::GeometryType& geo_type) {
414  std::ranges::copy(
415  std::views::iota(unsigned{0}, static_cast<unsigned int>(_points.size())),
416  std::back_inserter(_sorted_indices)
417  );
418  std::ranges::sort(_sorted_indices, [&] (unsigned int i1, unsigned int i2) {
419  const auto& key1 = _points[i1].localKey();
420  const auto& key2 = _points[i2].localKey();
421  using LagrangeDetail::dune_to_gfmt_sub_entity;
422  if (key1.codim() == key2.codim()) {
423  const auto mapped1 = dune_to_gfmt_sub_entity(geo_type, key1.subEntity(), key1.codim());
424  const auto mapped2 = dune_to_gfmt_sub_entity(geo_type, key2.subEntity(), key2.codim());
425  return mapped1 == mapped2 ? key1.index() < key2.index() : mapped1 < mapped2;
426  }
427  return _points[i1].localKey().codim() > _points[i2].localKey().codim();
428  });
429  }
430 
431  DUNE::EquidistantPointSet<typename GridView::ctype, GridView::dimension> _points;
432  ReservedVector<unsigned int, reserved_size> _sorted_indices;
433  };

571  auto _make_codim_mappers() {
572  _codim_to_mapper.clear();
573  _codim_to_mapper.emplace(0, Mapper{_grid_view, DUNE::mcmgLayout(DUNE::Codim<0>{})});
574  _codim_to_mapper.emplace(1, Mapper{_grid_view, DUNE::mcmgLayout(DUNE::Codim<1>{})});

577  if constexpr (int(GridView::dimension) == 3)

581  void _update_mesh() {
582  LagrangeDetail::PointIndicesHelper point_indices;
583  _cells.reserve(Traits::NumberOfCells<GridView>::get(_grid_view));
584  _element_seeds.reserve(_cells.size());
585  for (const auto& element : Traits::Cells<GridView>::get(_grid_view)) {
586  _element_to_running_index[_codim_to_mapper.at(0).index(element)] = _cells.size();
587  _element_seeds.push_back(element.seed());
588  auto& cell_points = _cells.emplace_back();
589  for (const auto& local_point : _local_points.at(element.type()).get()) {
590  const auto codim = local_point.localKey().codim();
591  const auto sub_entity = local_point.localKey().subEntity();
592  const auto sub_index = local_point.localKey().index();
593  const auto global_index = _codim_to_mapper.at(codim).subIndex(element, sub_entity, codim);
594  if (!point_indices.contains({codim, global_index, sub_index})) {
595  _points.emplace_back(element.geometry().global(local_point.point()));

717 template<typename Function, typename GridView, typename T = FunctionDetail::RangeScalar<Function, GridView>>
718 class FunctionField : public GridFormat::Field {
719  public:
720  explicit FunctionField(const Function& function,
721  const LagrangeMesh<GridView>& mesh,
722  const Precision<T>& = {},
723  bool cellwise_constant = false)
724  : _function{function}
725  , _mesh{mesh}
726  , _cellwise_constant{cellwise_constant} {
727  if constexpr (requires { function.basis(); }) {
728  static_assert(std::is_same_v<typename Function::Basis::GridView, GridView>);
729  if (&function.basis().gridView().grid() != &mesh.grid_view().grid())
730  throw ValueError("Function and mesh do not use the same underlying grid");
731  }
732  }
733 
734  private:
735  MDLayout _layout() const override {
736  return get_md_layout<FunctionDetail::RangeType<Function, GridView>>(
737  _cellwise_constant ? _mesh.number_of_cells()
738  : _mesh.number_of_points()
739  );
740  }
741 
742  DynamicPrecision _precision() const override {
743  return {Precision<T>{}};
744  }
745 
746  Serialization _serialized() const override {
747  const auto layout = _layout();
748  const auto num_entries = layout.number_of_entries();
749  const auto num_entries_per_value = layout.dimension() == 1 ? 1 : layout.number_of_entries(1);
750 
751  Serialization result(num_entries*sizeof(T));
752  auto out_data = result.template as_span_of<T>();
753  auto local_function = localFunction(_function);
754 
755  if (_cellwise_constant) {
756  std::size_t count = 0;
757  for (const auto& element : elements(_mesh.grid_view())) {
758  local_function.bind(element);
759  const auto& elem_geo = element.geometry();
760  const auto& local_pos = elem_geo.local(elem_geo.center());
761  std::size_t offset = (count++)*num_entries_per_value;
762  _copy_values(local_function(local_pos), out_data, offset);
763  }
764  } else {
765  for (const auto& element : elements(_mesh.grid_view())) {
766  local_function.bind(element);
767  for (const auto& point : _mesh.points(element)) {
768  const auto& local_pos = element.geometry().local(_mesh.position(point));
769  std::size_t offset = point.index*num_entries_per_value;
770  _copy_values(local_function(local_pos), out_data, offset);
771  }
772  }
773  }
774 
775  return result;
776  }
